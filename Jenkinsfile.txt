	  pipeline{
agent any
environment {
DEPLOY_CREDS = credentials('ANYPOINT')
}
stages {
   stage ('Build') {
    steps {
	echo 'Hello World'
	bat 'mvn -B -U -e -V clean -DskipTests package'
}
	  }
	  stage('Deploy CloudHub') {
	environment {
        ENVIRONMENT = 'Sandbox'
        APP_NAME = 'innerhttp2'
      }
	  steps {
           bat 'mvn -U -V -e -B -DskipTests deploy -DmuleDeploy -Danypoint.username="%DEPLOY_CREDS_USR%" -Danypoint.password="%DEPLOY_CREDS_PSW%"'
	  }
	  }
	   stage('Trigger Branch Build') {
      steps {
            script {
			        if (env.BRANCH_NAME == 'master')
                    echo "Triggering job for branch ${env.BRANCH_NAME}"
                    build job: "mudita.ti/Mulesoft_repository/master/dummy_job"
            }
        }
    }
	  }
	  }
